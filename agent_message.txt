def test_section_renderer_is_video(presentation):
    for s in get_content_sections(presentation):
        assert s.get("renderer") == "video", \
            f"Section {s['section_id']} has renderer={s.get('renderer')}, expected 'video'"

def test_every_segment_spec_has_renderer_reason(presentation):

        for spec in s.get("render_spec", {}).get("segment_specs", []):

                f"Section {s['section_id']} seg {spec.get('segment_id')} missing renderer_reason"

def test_timing_contract(presentation):
    for s in get_content_sections(presentation):
        specs = s.get("render_spec", {}).get("segment_specs", [])
        for i, spec in enumerate(specs):
            assert "start_seconds" in spec, f"Missing start_seconds in {spec.get('segment_id')}"
            assert "end_seconds" in spec, f"Missing end_seconds in {spec.get('segment_id')}"
            if i > 0:
                prev = specs[i-1]
                assert abs(spec["start_seconds"] - prev["end_seconds"]) < 0.1, \
